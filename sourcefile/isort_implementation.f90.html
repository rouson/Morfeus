SUBMODULE (tools_math) isort_implementation
    IMPLICIT NONE

    CONTAINS

        MODULE PROCEDURE isort
        USE psb_base_mod, ONLY : psb_msort, psb_sort_down_
        IMPLICIT NONE
        !! $Id: isort.f90 2469 2007-10-08 10:34:43Z sfilippo $
        !!
        !! Description:
        !!    Changed from bubble sort (YUCK!) to using PSBLAS sorting.
        !!
        INTEGER :: i, j, n, xx
        INTEGER, PARAMETER :: nth = 8

        n = SIZE(ivet)

        IF (n <= nth) THEN

            DO j = n - 1, 1, -1
                IF (ivet(j+1) > ivet(j)) THEN
                    xx = ivet(j)
                    i = j + 1
                    DO
                        ivet(i-1) = ivet(i)
                        i = i + 1
                        IF (i > n) EXIT
                        IF (ivet(i)  <=  xx) EXIT
                    ENDDO
                    ivet(i-1) = xx
                ENDIF
            ENDDO

        ELSE
            CALL psb_msort(ivet,dir=psb_sort_down_)
        END IF

        END PROCEDURE isort

END SUBMODULE isort_implementation
